\begin{frame}{System Model}
    \begin{columns}
        \begin{column}{0.58\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=\linewidth]{figure/preliminary/system_model.pdf}
                \caption{System model of \gls{dl}-based \gls{ntp}.}
            \end{figure}
        \end{column}
        \begin{column}{0.38\textwidth}
            \begin{itemize}
                \item A DNN is trained to forecast traffic at time $t$ from $T$ historical traffic snapshots $\delta^s(t-T), \ldots, \delta^s(t-1)$ (traffic demands at all base stations per slice $s$).
                \item \textbf{Input:} historical snapshots $\to$ 3D tensor (spatiotemporal).
                \item \textbf{Architecture:} encoder (3D-CNN) + decoder (MLP).
                \item \textbf{Output:} capacity forecast $c^s(t) = \{c^1_s(t), \ldots, c^M_s(t)\}$ at datacenters; custom loss balances overprovisioning vs.\ SLA violations (tuned by $\alpha = \beta/\gamma$).
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

    \begin{itemize}
        \item Adversary controls \alert{compromised devices} (\eg, IoT, smartphones) that inject traffic into the network (Mirai-style).
        \item Goal: the \gls{dnn} \alert{over-/under-provisioning} $\rightarrow$ wasteful/service degradation.
        \item \alert{Injection is minimal} to avoid detection and stay within data limits.
    \end{itemize}
\end{frame}

\begin{frame}{Local Robustness Properties for DL-based NTP (1)}
    \begin{block}{Input property $\phi_{in}$}
        Encodes \emph{permitted adversarial traffic injection} as a hyperrectangle.
        \begin{itemize}
            \item Perturbation $\eta \in [\eta_L, \eta_U]$ (same dim as traffic snapshot $\delta^s(t)$).
            \item Adversarial input: $\delta'^s(t) = \delta^s(t) + \eta$.
            \item $\phi_{in}$ constrains perturbed traffic to $[\delta^s(t) + \eta_L, \delta^s(t) + \eta_U]$ so verification covers all injection scenarios within bounds.
        \end{itemize}
    \end{block}
\end{frame}

\begin{frame}{Local Robustness Properties for DL-based NTP (2)}
    \begin{block}{Output property $\phi_{out}$}
        Constrain forecast vs.\ actual demand $d^j_s(t)$ (tolerance $\zeta$):
        \begin{itemize}
            \item \textbf{Overprovisioning:} $c^j_s(t) \leq (1+\zeta)\, d^j_s(t)$.
            \item \textbf{Underprovisioning:} $c^j_s(t) \geq (1-\zeta)\, d^j_s(t)$.
        \end{itemize}
    \end{block}
    \begin{block}{Verification outcome}
        \textbf{SAT:} exists adversarial input in $[\eta_L, \eta_U]$ violating $\phi_{out}$ (vulnerable). \quad \textbf{UNSAT:} no such input $\Rightarrow$ \emph{proven robust}.
    \end{block}
\end{frame}
