\section{Preliminaries}

\begin{frame}{System Model}
    \begin{columns}
        \begin{column}{0.58\textwidth}
            \begin{figure}
                \centering
                \includegraphics[width=\linewidth]{figure/preliminary/system_model.pdf}
                \caption{System model of \gls{dl}-based \gls{ntp}.}
            \end{figure}
        \end{column}
        \begin{column}{0.38\textwidth}
            \begin{itemize}
                \item A DNN is trained to forecast traffic at time $t$ from $T$ historical traffic snapshots $\delta^s(t-T), \ldots, \delta^s(t-1)$ (traffic demands at all base stations per slice $s$).
                \item \textbf{Input:} historical snapshots $\to$ 3D tensor (spatiotemporal).
                \item \textbf{Architecture:} encoder (3D-CNN) + decoder (MLP).
                \item \textbf{Output:} capacity forecast $c^s(t) = \{c^1_s(t), \ldots, c^M_s(t)\}$ at datacenters; custom loss balances overprovisioning vs.\ SLA violations (tuned by $\alpha = \beta/\gamma$).
            \end{itemize}
        \end{column}
    \end{columns}
\end{frame}

\begin{frame}{Threat Model}
    \begin{figure}
        \centering
        \includegraphics[width=0.7\linewidth]{figure/preliminary/threat_model.pdf}
        \caption{Threat model of \gls{dl}-based \gls{ntp}.}

    \begin{itemize}
        \item Adversary controls \alert{compromised devices} (\eg, IoT, smartphones) that inject traffic into the network (Mirai-style).
        \item Goal: the \gls{dnn} \alert{over-/under-provisioning} $\rightarrow$ wasteful/service degradation.
        \item \alert{Injection is minimal} to avoid detection and stay within data limits.
    \end{itemize}
\end{frame}
