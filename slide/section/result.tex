\section{Evaluation Results}

\begin{frame}{Evaluation Setup}
    \begin{itemize}
        \item \textbf{Dataset:} Telecom Italia Milan (2014), 1,728 base stations $\to$ $\sim$10k cells; 10-minute granularity; Internet activities as traffic proxy~\cite{bega2019deepcog}.
        \item \textbf{Model:} DeepCog; Adam ($3\times10^{-4}$), 50 epochs; $5\times5$ grid, central-cell forecast; $\alpha=2$ (prioritize avoiding SLA violations).
        \item \textbf{Metrics:} For each instance, \texttt{NeuralSAT} returns \texttt{SAT} (vulnerability), \texttt{UNSAT} (certified robust), or \texttt{TIMEOUT}; proportions over injection levels $\eta$ and tolerance $\zeta$.
    \end{itemize}
\end{frame}

\begin{frame}{Verification Results -- Underestimation vs.\ Overestimation}
    \begin{columns}
        \begin{column}{0.48\textwidth}
            \begin{block}{Underestimation (SLA violation)}
                Vary $\zeta \in \{1\%,5\%,10\%\}$, injection +1\%--+20\%.
                \textbf{Result:} Consistently \texttt{SAT} across all cells $\Rightarrow$ model \emph{vulnerable}; adversary can force capacity underestimation beyond $\zeta$.
            \end{block}
        \end{column}
        \begin{column}{0.48\textwidth}
            \begin{block}{Overestimation (overprovisioning)}
                Same $\zeta$ and injection levels.
                \textbf{Result:} Mostly \texttt{UNSAT} at +1\%, +5\%, +10\% $\Rightarrow$ \emph{provably robust}; at +15\%, +20\% more \texttt{SAT}/\texttt{TIMEOUT}.
            \end{block}
        \end{column}
    \end{columns}
    \vspace{0.5em}
    \begin{alertblock}{Takeaway}
        Asymmetric robustness: provably resilient to overestimation under moderate perturbations; highly vulnerable to underestimation attacks.
    \end{alertblock}
\end{frame}

\begin{frame}{Result Tables (UNSAT / SAT / TIMEOUT)}
    \small
    \textbf{Underestimation} (fraction UNSAT/SAT/TIMEOUT): all cells $0.0/1.0/0.0$ (all \texttt{SAT}). \\
    \textbf{Overestimation}: e.g.\ $\zeta=1\%$: $1.0/0.0/0.0$ for +1--+10\%; $0.0/1.0/0.0$ at +15\%; $\zeta=10\%$: $1.0/0.0/0.0$ for +1--+15\%, $0.0/0.0/1.0$ at +20\%.

    \end{center}
\end{frame}
