\section{Evaluation Results}

\begin{frame}{Evaluation Setup}
    \begin{block}{Dataset --- Telecom Italia Milan 2014}
        \begin{itemize}
            \item Publicly available mobile traffic data; 1,728 base stations aggregated into $\sim$10k cells (Voronoi).
            \item $\sim 7000$ time steps, $~50$ days measurements, 10-minute granularity.
            \item \emph{Internet activities} traffic demands~\cite{bega2019deepcog}.
            \item Spatial-temporal pattern well-suited for DeepCog-based traffic forecasting.
        \end{itemize}
    \end{block}
    \begin{block}{DNN Parameter --- DeepCog}
        \begin{itemize}
            \item $5\times5$ grid; $T=10$ historical steps; each model forecasts capacity of the \emph{central cell} only~\cite{gholian2025deexp}.
            \item $\alpha=2$ (prioritize avoiding SLA violations over overprovisioning)~\cite{bega2019deepcog}.
            \item Adam, learning rate $3\times10^{-4}$, 50 epochs; ReLU activation; 80:20 train-test split.
            \item Verify on perturbed data on test set, \neuralsat timeout after $60$s.
        \end{itemize}
        \footnote{Publicly available implementation at \url{https://github.com/thanhlexyz/nnv4ntp}}
    \end{block}
\end{frame}

    \begin{block}{Observations}
        \begin{itemize}
            \item \textbf{Small injection} (+1\%, +5\%, +10\%): \texttt{NeuralSAT} often \textcolor{green}{\texttt{UNSAT}} $\Rightarrow$ \emph{provably robust} against overprovisioning.
            \item \textbf{High injection} (+15\%, +20\%): more \texttt{SAT}/\texttt{TIMEOUT} $\Rightarrow$ extreme perturbations can compromise robustness.
            \item Takeaway: asymmetric profile --- resilient to overestimation under moderate $\eta$; highly vulnerable to underestimation.
        \end{itemize}
    \end{block}
